```yaml
# SMYKM Campaign Pipeline — GitHub Actions Workflow
# Phase 0 skeleton: trigger-to-runner path only. No script logic yet.
# Phase 1 will replace placeholder echo steps with real script calls.

name: SMYKM Campaign Pipeline

on:
  workflow_dispatch:
    inputs:
      contact_id:
        description: 'HubSpot Contact ID'
        required: true
        type: string
      contact_email:
        description: 'Contact email address'
        required: true
        type: string

# INFRA-04: Per-contact concurrency group prevents duplicate runs for the same contact.
# GitHub Actions concurrency cannot cap at exactly N simultaneous runs (platform limitation).
# Design decision: unique group per contact_id allows different contacts to run in parallel
# (up to GitHub free-tier limit of 20 concurrent), while preventing race conditions when
# Make.com fires the same contact twice. cancel-in-progress: false ensures the second
# dispatch queues rather than being silently dropped.
concurrency:
  group: campaign-${{ github.event.inputs.contact_id }}
  cancel-in-progress: false

defaults:
  run:
    working-directory: SMYKM/

jobs:
  generate-campaign:
    name: Generate SMYKM Campaign
    runs-on: ubuntu-latest

    steps:
      # ── Checkout ──────────────────────────────────────────────────────────
      - name: Checkout repository
        uses: actions/checkout@v4
        # NOTE: uses: steps ignore defaults.run.working-directory (platform behaviour).
        # Checkout always runs from repo root. This is expected.

      # ── Inputs echo (safe — contact_id and contact_email are NOT secrets) ─
      - name: Echo dispatch inputs
        timeout-minutes: 10
        run: |
          echo "contact_id=${{ github.event.inputs.contact_id }}"
          echo "contact_email=${{ github.event.inputs.contact_email }}"

      # ── Secret verification (INFRA-03) ────────────────────────────────────
      # Verifies all four secrets are non-empty. Logs length only — never the value.
      # ${{ secrets.X }} evaluates to empty string if the secret does not exist.
      - name: Verify all secrets are loaded
        timeout-minutes: 10
        env:
          HUBSPOT_API_KEY: ${{ secrets.HUBSPOT_API_KEY }}
          CHORUS_API_TOKEN: ${{ secrets.CHORUS_API_TOKEN }}
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
          TEAMS_WEBHOOK_URL: ${{ secrets.TEAMS_WEBHOOK_URL }}
        run: |
          for var in HUBSPOT_API_KEY CHORUS_API_TOKEN ANTHROPIC_API_KEY TEAMS_WEBHOOK_URL; do
            val="${!var}"
            if [ -z "$val" ]; then
              echo "ERROR: $var is empty or not set" && exit 1
            fi
            echo "$var is set (length: ${#val})"
          done

      # ── Working directory verification (INFRA-06) ─────────────────────────
      - name: Verify working directory
        timeout-minutes: 10
        run: |
          echo "Working dir: $(pwd)"
          ls -la

      # ── Python setup ──────────────────────────────────────────────────────
      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.12'

      - name: Install dependencies
        timeout-minutes: 10
        run: pip install -r requirements.txt

      # ── Script placeholders (Phase 0 skeletons — replaced in Phase 1) ─────
      - name: '[PLACEHOLDER] Fetch HubSpot data'
        timeout-minutes: 10
        env:
          HUBSPOT_API_KEY: ${{ secrets.HUBSPOT_API_KEY }}
          CONTACT_ID: ${{ github.event.inputs.contact_id }}
          CONTACT_EMAIL: ${{ github.event.inputs.contact_email }}
        run: |
          echo "PLACEHOLDER: fetch_hubspot.py will run here in Phase 2"
          echo "CONTACT_ID=$CONTACT_ID"

      - name: '[PLACEHOLDER] Fetch Chorus transcripts'
        timeout-minutes: 10
        env:
          CHORUS_API_TOKEN: ${{ secrets.CHORUS_API_TOKEN }}
        run: |
          echo "PLACEHOLDER: fetch_chorus.py will run here in Phase 2"

      - name: '[PLACEHOLDER] Compute campaign tokens'
        timeout-minutes: 10
        run: |
          echo "PLACEHOLDER: compute_campaign_tokens.py will run here in Phase 2"

      - name: '[PLACEHOLDER] Generate campaign via Claude'
        timeout-minutes: 10
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: |
          echo "PLACEHOLDER: generate_campaign.py will run here in Phase 3"

      - name: '[PLACEHOLDER] Write campaign to HubSpot'
        timeout-minutes: 10
        env:
          HUBSPOT_API_KEY: ${{ secrets.HUBSPOT_API_KEY }}
        run: |
          echo "PLACEHOLDER: write_hubspot.py will run here in Phase 4"

      # ── Artifact upload: campaign output (OBS-01 — always, 7-day) ─────────
      # NOTE: Phase 0 — no campaign_output.json exists yet.
      # if-no-files-found: ignore prevents step failure. Activated in Phase 5.
      - name: Upload campaign output artifact
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: campaign-output-${{ github.run_id }}-${{ github.run_attempt }}
          path: ${{ runner.temp }}/campaign_output.json
          retention-days: 7
          if-no-files-found: ignore

      # ── Failure path: DLQ artifact (ERR-05 — only on failure) ─────────────
      - name: Upload DLQ artifact
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: failed-contacts-${{ github.run_id }}-${{ github.run_attempt }}
          path: ${{ runner.temp }}/failed_contacts.json
          retention-days: 30
          if-no-files-found: ignore

      # ── Failure path: Teams/Slack notification (ERR-06) ───────────────────
      - name: '[PLACEHOLDER] Send failure notification'
        if: failure()
        env:
          TEAMS_WEBHOOK_URL: ${{ secrets.TEAMS_WEBHOOK_URL }}
          CONTACT_EMAIL: ${{ github.event.inputs.contact_email }}
        run: |
          echo "PLACEHOLDER: Teams/Slack notification will be implemented in Phase 1"
          echo "Failed contact: $CONTACT_EMAIL"
          echo "Run URL: ${{ github.server_url }}/${{ github.repository }}/actions/runs/${{ github.run_id }}"
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| workflow_dispatch inputs → runner env | contact_id and contact_email arrive from Make.com (external) or manual UI trigger; treated as untrusted strings |
| GitHub secrets → step env | Injected at runtime by GitHub; auto-masked in logs; never in YAML source |
| Runner env → bash run: steps | Env vars read by bash; no eval or dynamic execution in this phase |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-00-01 | Information Disclosure | Secret values in step logs | mitigate | Inject secrets via env: blocks only; never use ${{ secrets.X }} inline in run: commands. Secret verification step logs length only, not value. GitHub auto-masking provides defence-in-depth. |
| T-00-02 | Tampering | CRLF injection via Windows commits breaking bash steps | mitigate | .gitattributes sets *.yml text eol=lf before any .yml file is committed. |
| T-00-03 | Denial of Service | Duplicate Make.com dispatches for same contact creating race conditions | mitigate | concurrency.group: campaign-${{ inputs.contact_id }} with cancel-in-progress: false queues duplicates rather than racing or silently dropping. |
| T-00-04 | Elevation of Privilege | Fine-grained PAT with excessive scopes | mitigate | PAT scoped to Actions: write only on the single SMYKM repo. Not Contents: write (which would allow code push). Documented in Plan 02. |
| T-00-05 | Information Disclosure | contact_id / contact_email visible in run UI | accept | workflow_dispatch inputs are not secrets (prospect IDs and emails). They are intentionally visible in the GitHub UI for debugging. This is documented behaviour, not a defect. |
| T-00-06 | Denial of Service | Hung steps consuming runner slots | mitigate | timeout-minutes: 10 on every run: step; job automatically cancelled at step timeout. |
</threat_model>
